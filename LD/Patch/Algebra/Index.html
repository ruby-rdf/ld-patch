def execute(queryable, options = {})
  debug(options) {"Index"}
  terms = Array(options.fetch(:terms))
  index = operand(0)

  results = terms.map do |term|
    list = RDF::List.new(subject: term, graph: queryable)
    list.at(index.to_i)
  end.flatten

  RDF::Query::Solutions.new(results.map {|t| RDF::Query::Solution.new(path: t)})
end
